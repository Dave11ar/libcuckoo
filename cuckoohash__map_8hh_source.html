1198  // cuckoo_insert tries to find an empty slot in either of the buckets to
1199  // insert the given key into, performing cuckoo hashing if necessary. It
1200  // expects the locks to be taken outside the function. Before inserting, it
1201  // checks that the key isn't already in the table. cuckoo hashing presents
1202  // multiple concurrency issues, which are explained in the function. The
1203  // following return states are possible:
1204  //
1205  // ok -- Found an empty slot, locks will be held on both buckets after the
1206  // function ends, and the position of the empty slot is returned
1207  //
1208  // failure_key_duplicated -- Found a duplicate key, locks will be held, and
1209  // the position of the duplicate key will be returned
1210  //
1211  // failure_under_expansion -- Failed due to a concurrent expansion
1212  // operation. Locks are released. No meaningful position is returned.
1213  //
1214  // failure_table_full -- Failed to find an empty slot for the table. Locks
1215  // are released. No meaningful position is returned.
1216  template <typename TABLE_MODE, typename K>
1217  table_position cuckoo_insert(const hash_value hv, TwoBuckets &b, K &key) {
1218  int res1, res2;
1219  bucket &b1 = buckets_[b.i1];
1220  if (!try_find_insert_bucket(b1, res1, hv.partial, key)) {
1221  return table_position{b.i1, static_cast<size_type>(res1),
1222  failure_key_duplicated};
1223  }
1224  bucket &b2 = buckets_[b.i2];
1225  if (!try_find_insert_bucket(b2, res2, hv.partial, key)) {
1226  return table_position{b.i2, static_cast<size_type>(res2),
1227  failure_key_duplicated};
1228  }
1229  if (res1 != -1) {
1230  return table_position{b.i1, static_cast<size_type>(res1), ok};
1231  }
1232  if (res2 != -1) {
1233  return table_position{b.i2, static_cast<size_type>(res2), ok};
1234  }
1235 
1236  // We are unlucky, so let's perform cuckoo hashing.
1237  size_type insert_bucket = 0;
1238  size_type insert_slot = 0;
1239  cuckoo_status st = run_cuckoo<TABLE_MODE>(b, insert_bucket, insert_slot);
1240  if (st == failure_under_expansion) {
1241  // The run_cuckoo operation operated on an old version of the table,
1242  // so we have to try again. We signal to the calling insert method
1243  // to try again by returning failure_under_expansion.
1244  return table_position{0, 0, failure_under_expansion};
1245  } else if (st == ok) {
1246  assert(TABLE_MODE() == locked_table_mode() ||
1247  !get_current_locks()[lock_ind(b.i1)].try_lock());
1248  assert(TABLE_MODE() == locked_table_mode() ||
1249  !get_current_locks()[lock_ind(b.i2)].try_lock());
1250  assert(!buckets_[insert_bucket].occupied(insert_slot));
1251  assert(insert_bucket == index_hash(hashpower(), hv.hash) ||
1252  insert_bucket == alt_index(hashpower(), hv.partial,
1253  index_hash(hashpower(), hv.hash)));
1254  // Since we unlocked the buckets during run_cuckoo, another insert
1255  // could have inserted the same key into either b.i1 or
1256  // b.i2, so we check for that before doing the insert.
1257  table_position pos = cuckoo_find(key, hv.partial, b.i1, b.i2);
1258  if (pos.status == ok) {
1259  pos.status = failure_key_duplicated;
1260  return pos;
1261  }
1262  return table_position{insert_bucket, insert_slot, ok};
1263  }
1264  assert(st == failure);
1265  LIBCUCKOO_DBG("hash table is full (hashpower = %zu, hash_items = %zu,"
1266  "load factor = %.2f), need to increase hashpower\n",
1267  hashpower(), size(), load_factor());
1268  return table_position{0, 0, failure_table_full};
1269  }

1326  // run_cuckoo performs cuckoo hashing on the table in an attempt to free up
1327  // a slot on either of the insert buckets, which are assumed to be locked
1328  // before the start. On success, the bucket and slot that was freed up is
1329  // stored in insert_bucket and insert_slot. In order to perform the search
1330  // and the swaps, it has to release the locks, which can lead to certain
1331  // concurrency issues, the details of which are explained in the function.
1332  // If run_cuckoo returns ok (success), then `b` will be active, otherwise it
1333  // will not.
1334  template <typename TABLE_MODE>
1335  cuckoo_status run_cuckoo(TwoBuckets &b, size_type &insert_bucket,
1336  size_type &insert_slot) {
1337  // We must unlock the buckets here, so that cuckoopath_search and
1338  // cuckoopath_move can lock buckets as desired without deadlock.
1339  // cuckoopath_move has to move something out of one of the original
1340  // buckets as its last operation, and it will lock both buckets and
1341  // leave them locked after finishing. This way, we know that if
1342  // cuckoopath_move succeeds, then the buckets needed for insertion are
1343  // still locked. If cuckoopath_move fails, the buckets are unlocked and
1344  // we try again. This unlocking does present two problems. The first is
1345  // that another insert on the same key runs and, finding that the key
1346  // isn't in the table, inserts the key into the table. Then we insert
1347  // the key into the table, causing a duplication. To check for this, we
1348  // search the buckets for the key we are trying to insert before doing
1349  // so (this is done in cuckoo_insert, and requires that both buckets are
1350  // locked). Another problem is that an expansion runs and changes the
1351  // hashpower, meaning the buckets may not be valid anymore. In this
1352  // case, the cuckoopath functions will have thrown a hashpower_changed
1353  // exception, which we catch and handle here.
1354  size_type hp = hashpower();
1355  b.unlock();
1356  CuckooRecords cuckoo_path;
1357  bool done = false;
1358  try {
1359  while (!done) {
1360  const int depth =
1361  cuckoopath_search<TABLE_MODE>(hp, cuckoo_path, b.i1, b.i2);
1362  if (depth < 0) {
1363  break;
1364  }
1365 
1366  if (cuckoopath_move<TABLE_MODE>(hp, cuckoo_path, depth, b)) {
1367  insert_bucket = cuckoo_path[0].bucket;
1368  insert_slot = cuckoo_path[0].slot;
1369  assert(insert_bucket == b.i1 || insert_bucket == b.i2);
1370  assert(TABLE_MODE() == locked_table_mode() ||
1371  !get_current_locks()[lock_ind(b.i1)].try_lock());
1372  assert(TABLE_MODE() == locked_table_mode() ||
1373  !get_current_locks()[lock_ind(b.i2)].try_lock());
1374  assert(!buckets_[insert_bucket].occupied(insert_slot));
1375  done = true;
1376  break;
1377  }
1378  }
1379  } catch (hashpower_changed &) {
1380  // The hashpower changed while we were trying to cuckoo, which means
1381  // we want to retry. b.i1 and b.i2 should not be locked
1382  // in this case.
1383  return failure_under_expansion;
1384  }
1385  return done ? ok : failure;
1386  }

1388  // cuckoopath_search finds a cuckoo path from one of the starting buckets to
1389  // an empty slot in another bucket. It returns the depth of the discovered
1390  // cuckoo path on success, and -1 on failure. Since it doesn't take locks on
1391  // the buckets it searches, the data can change between this function and
1392  // cuckoopath_move. Thus cuckoopath_move checks that the data matches the
1393  // cuckoo path before changing it.
1394  //
1395  // throws hashpower_changed if it changed during the search.
1396  template <typename TABLE_MODE>
1397  int cuckoopath_search(const size_type hp, CuckooRecords &cuckoo_path,
1398  const size_type i1, const size_type i2) {
1399  b_slot x = slot_search<TABLE_MODE>(hp, i1, i2);
1400  if (x.depth == -1) {
1401  return -1;
1402  }
1403  // Fill in the cuckoo path slots from the end to the beginning.
1404  for (int i = x.depth; i >= 0; i--) {
1405  cuckoo_path[i].slot = x.pathcode % slot_per_bucket();
1406  x.pathcode /= slot_per_bucket();
1407  }
1408  // Fill in the cuckoo_path buckets and keys from the beginning to the
1409  // end, using the final pathcode to figure out which bucket the path
1410  // starts on. Since data could have been modified between slot_search
1411  // and the computation of the cuckoo path, this could be an invalid
1412  // cuckoo_path.
1413  CuckooRecord &first = cuckoo_path[0];
1414  if (x.pathcode == 0) {
1415  first.bucket = i1;
1416  } else {
1417  assert(x.pathcode == 1);
1418  first.bucket = i2;
1419  }
1420  {
1421  const auto lock_manager = lock_one(hp, first.bucket, TABLE_MODE());
1422  const bucket &b = buckets_[first.bucket];
1423  if (!b.occupied(first.slot)) {
1424  // We can terminate here
1425  return 0;
1426  }
1427  first.hv = hashed_key(b.key(first.slot));
1428  }
1429  for (int i = 1; i <= x.depth; ++i) {
1430  CuckooRecord &curr = cuckoo_path[i];
1431  const CuckooRecord &prev = cuckoo_path[i - 1];
1432  assert(prev.bucket == index_hash(hp, prev.hv.hash) ||
1433  prev.bucket ==
1434  alt_index(hp, prev.hv.partial, index_hash(hp, prev.hv.hash)));
1435  // We get the bucket that this slot is on by computing the alternate
1436  // index of the previous bucket
1437  curr.bucket = alt_index(hp, prev.hv.partial, prev.bucket);
1438  const auto lock_manager = lock_one(hp, curr.bucket, TABLE_MODE());
1439  const bucket &b = buckets_[curr.bucket];
1440  if (!b.occupied(curr.slot)) {
1441  // We can terminate here
1442  return i;
1443  }
1444  curr.hv = hashed_key(b.key(curr.slot));
1445  }
1446  return x.depth;
1447  }

1449  // cuckoopath_move moves keys along the given cuckoo path in order to make
1450  // an empty slot in one of the buckets in cuckoo_insert. Before the start of
1451  // this function, the two insert-locked buckets were unlocked in run_cuckoo.
1452  // At the end of the function, if the function returns true (success), then
1453  // both insert-locked buckets remain locked. If the function is
1454  // unsuccessful, then both insert-locked buckets will be unlocked.
1455  //
1456  // throws hashpower_changed if it changed during the move.
1457  template <typename TABLE_MODE>
1458  bool cuckoopath_move(const size_type hp, CuckooRecords &cuckoo_path,
1459  size_type depth, TwoBuckets &b) {
1460  if (depth == 0) {
1461  // There is a chance that depth == 0, when try_add_to_bucket sees
1462  // both buckets as full and cuckoopath_search finds one empty. In
1463  // this case, we lock both buckets. If the slot that
1464  // cuckoopath_search found empty isn't empty anymore, we unlock them
1465  // and return false. Otherwise, the bucket is empty and insertable,
1466  // so we hold the locks and return true.
1467  const size_type bucket_i = cuckoo_path[0].bucket;
1468  assert(bucket_i == b.i1 || bucket_i == b.i2);
1469  b = lock_two(hp, b.i1, b.i2, TABLE_MODE());
1470  if (!buckets_[bucket_i].occupied(cuckoo_path[0].slot)) {
1471  return true;
1472  } else {
1473  b.unlock();
1474  return false;
1475  }
1476  }
1477 
1478  while (depth > 0) {
1479  CuckooRecord &from = cuckoo_path[depth - 1];
1480  CuckooRecord &to = cuckoo_path[depth];
1481  const size_type fs = from.slot;
1482  const size_type ts = to.slot;
1483  TwoBuckets twob;
1484  LockManager extra_manager;
1485  if (depth == 1) {
1486  // Even though we are only swapping out of one of the original
1487  // buckets, we have to lock both of them along with the slot we
1488  // are swapping to, since at the end of this function, they both
1489  // must be locked. We store tb inside the extrab container so it
1490  // is unlocked at the end of the loop.
1491  std::tie(twob, extra_manager) =
1492  lock_three(hp, b.i1, b.i2, to.bucket, TABLE_MODE());
1493  } else {
1494  twob = lock_two(hp, from.bucket, to.bucket, TABLE_MODE());
1495  }

1532  struct b_slot {
1533  // The bucket of the last item in the path.
1534  size_type bucket;

1548  std::numeric_limits<decltype(depth)>::max(),
1549  "The depth type must able to hold a value of"
1550  " MAX_BFS_PATH_LEN - 1");
1551  static_assert(-1 >= std::numeric_limits<decltype(depth)>::min(),
1552  "The depth type must be able to hold a value of -1");

1560  // b_queue is the queue used to store b_slots for BFS cuckoo hashing.
1561  class b_queue {
1562  public:
1563  b_queue() noexcept : first_(0), last_(0) {}
1564 
1565  void enqueue(b_slot x) {
1566  assert(!full());
1567  slots_[last_++] = x;
1568  }
1569 
1570  b_slot dequeue() {
1571  assert(!empty());
1572  assert(first_ < last_);
1573  b_slot &x = slots_[first_++];
1574  return x;
1575  }
1576 
1577  bool empty() const { return first_ == last_; }
1578 
1579  bool full() const { return last_ == MAX_CUCKOO_COUNT; }
1580 
1581  private:
1582  // The size of the BFS queue. It holds just enough elements to fulfill a
1583  // MAX_BFS_PATH_LEN search for two starting buckets, with no circular
1584  // wrapping-around. For one bucket, this is the geometric sum
1585  // sum_{k=0}^{MAX_BFS_PATH_LEN-1} slot_per_bucket()^k
1586  // = (1 - slot_per_bucket()^MAX_BFS_PATH_LEN) / (1 - slot_per_bucket())
1587  //
1588  // Note that if slot_per_bucket() == 1, then this simply equals
1589  // MAX_BFS_PATH_LEN.
1590  static_assert(slot_per_bucket() > 0,
1591  "SLOT_PER_BUCKET must be greater than 0.");
1592  static constexpr size_type MAX_CUCKOO_COUNT =
1593  2 * ((slot_per_bucket() == 1)
1594  ? MAX_BFS_PATH_LEN
1595  : (const_pow(slot_per_bucket(), MAX_BFS_PATH_LEN) - 1) /
1596  (slot_per_bucket() - 1));
1597  // An array of b_slots. Since we allocate just enough space to complete a
1598  // full search, we should never exceed the end of the array.
1599  b_slot slots_[MAX_CUCKOO_COUNT];
1600  // The index of the head of the queue in the array
1601  size_type first_;
1602  // One past the index of the last_ item of the queue in the array.
1603  size_type last_;
1604  };

1606  // slot_search searches for a cuckoo path using breadth-first search. It
1607  // starts with the i1 and i2 buckets, and, until it finds a bucket with an
1608  // empty slot, adds each slot of the bucket in the b_slot. If the queue runs
1609  // out of space, it fails.
1610  //
1611  // throws hashpower_changed if it changed during the search
1612  template <typename TABLE_MODE>
1613  b_slot slot_search(const size_type hp, const size_type i1,
1614  const size_type i2) {
1615  b_queue q;
1616  // The initial pathcode informs cuckoopath_search which bucket the path
1617  // starts on
1618  q.enqueue(b_slot(i1, 0, 0));
1619  q.enqueue(b_slot(i2, 1, 0));
1620  while (!q.empty()) {
1621  b_slot x = q.dequeue();
1622  auto lock_manager = lock_one(hp, x.bucket, TABLE_MODE());
1623  bucket &b = buckets_[x.bucket];
1624  // Picks a (sort-of) random slot to start from
1625  size_type starting_slot = x.pathcode % slot_per_bucket();
1626  for (size_type i = 0; i < slot_per_bucket(); ++i) {
1627  uint16_t slot = (starting_slot + i) % slot_per_bucket();
1628  if (!b.occupied(slot)) {
1629  // We can terminate the search here
1630  x.pathcode = x.pathcode * slot_per_bucket() + slot;
1631  return x;
1632  }
1633 
1634  // If x has less than the maximum number of path components,
1635  // create a new b_slot item, that represents the bucket we would
1636  // have come from if we kicked out the item at this slot.
1637  const partial_t partial = b.partial(slot);
1638  if (x.depth < MAX_BFS_PATH_LEN - 1) {
1639  assert(!q.full());
1640  b_slot y(alt_index(hp, partial, x.bucket),
1641  x.pathcode * slot_per_bucket() + slot, x.depth + 1);
1642  q.enqueue(y);
1643  }
1644  }
1645  }
1646  // We didn't find a short-enough cuckoo path, so the search terminated.
1647  // Return a failure value.
1648  return b_slot(0, 0, -1);
1649  }

1651  // cuckoo_fast_double will double the size of the table by taking advantage
1652  // of the properties of index_hash and alt_index. If the key's move
1653  // constructor is not noexcept, we use cuckoo_expand_simple, since that
1654  // provides a strong exception guarantee.
1655  template <typename TABLE_MODE, typename AUTO_RESIZE>
1656  cuckoo_status cuckoo_fast_double(size_type current_hp) {
1657  if (!is_data_nothrow_move_constructible()) {
1658  LIBCUCKOO_DBG("%s", "cannot run cuckoo_fast_double because key-value"
1659  " pair is not nothrow move constructible");
1660  return cuckoo_expand_simple<TABLE_MODE, AUTO_RESIZE>(current_hp + 1);
1661  }
1662  const size_type new_hp = current_hp + 1;
1663  auto all_locks_manager = lock_all(TABLE_MODE());
1664  cuckoo_status st = check_resize_validity<AUTO_RESIZE>(current_hp, new_hp);
1665  if (st != ok) {
1666  return st;
1667  }
1668 
1669  // Finish rehashing any un-rehashed buckets, so that we can move out any
1670  // remaining data in old_buckets_. We should be running cuckoo_fast_double
1671  // only after trying to cuckoo for a while, which should mean we've tried
1672  // going through most of the table and thus done a lot of rehashing
1673  // already. So this shouldn't be too expensive.
1674  //
1675  // We restrict ourselves to the current thread because we want to avoid
1676  // possibly spawning extra threads in this function, unless the
1677  // circumstances are predictable (i.e. data is nothrow move constructible,
1678  // we're in locked_table mode and must keep the buckets_ container
1679  // up-to-date, etc).
1680  //
1681  // If we have fewer than kNumLocks buckets, there shouldn't be any buckets
1682  // left to rehash, so this should be a no-op.
1683  {
1684  locks_t &current_locks = get_current_locks();
1685  for (size_t i = 0; i < current_locks.size(); ++i) {
1686  rehash_lock<kIsNotLazy>(i);
1687  }
1688  num_remaining_lazy_rehash_locks(0);
1689  }
1690 
1691  // Resize the locks array if necessary. This is done before we update the
1692  // hashpower so that other threads don't grab the new hashpower and the old
1693  // locks.
1694  maybe_resize_locks(size_type(1) << new_hp);
1695  locks_t &current_locks = get_current_locks();
1696 
1697  // Move the current buckets into old_buckets_, and create a new empty
1698  // buckets container, which will become the new current one. The
1699  // old_buckets_ data will be destroyed when move-assigning to buckets_.
1700  old_buckets_.swap(buckets_);
1701  buckets_ = buckets_t(new_hp, get_allocator());
1702 
1703  // If we have less than kMaxNumLocks buckets, we do a full rehash in the
1704  // current thread. On-demand rehashing wouldn't be very easy with less than
1705  // kMaxNumLocks buckets, because it would require taking extra lower-index
1706  // locks to do the rehashing. Because kMaxNumLocks is relatively small,
1707  // this should not be very expensive. We have already set all locks to
1708  // migrated at the start of the function, so we shouldn't have to touch
1709  // them again.
1710  //
1711  // Otherwise, if we're in locked_table_mode, the expectation is that we can
1712  // access the latest data in buckets_ without taking any locks. So we must
1713  // rehash the data immediately. This would not be much different from
1714  // lazy-rehashing in locked_table_mode anyways, because it would still be
1715  // going on in one thread.
1716  if (old_buckets_.size() < kMaxNumLocks) {
1717  for (size_type i = 0; i < old_buckets_.size(); ++i) {
1718  move_bucket(old_buckets_, buckets_, i);
1719  }
1720  // This will also delete the old_buckets_ data.
1721  num_remaining_lazy_rehash_locks(0);
1722  } else {
1723  // Mark all current locks as un-migrated, so that we rehash the data
1724  // on-demand when the locks are taken.
1725  for (spinlock &lock : current_locks) {
1726  lock.is_migrated() = false;
1727  }
1728  num_remaining_lazy_rehash_locks(current_locks.size());
1729  if (std::is_same<TABLE_MODE, locked_table_mode>::value) {
1730  rehash_with_workers();
1731  }
1732  }
1733  return ok;
1734  }

1736  void move_bucket(buckets_t &old_buckets, buckets_t &new_buckets,
1737  size_type old_bucket_ind) const noexcept {
1738  const size_t old_hp = old_buckets.hashpower();
1739  const size_t new_hp = new_buckets.hashpower();
1740 
1741  // By doubling the table size, the index_hash and alt_index of each key got
1742  // one bit added to the top, at position old_hp, which means anything we
1743  // have to move will either be at the same bucket position, or exactly
1744  // hashsize(old_hp) later than the current bucket.
1745  bucket &old_bucket = old_buckets_[old_bucket_ind];
1746  const size_type new_bucket_ind = old_bucket_ind + hashsize(old_hp);
1747  size_type new_bucket_slot = 0;
1748 
1749  // For each occupied slot, either move it into its same position in the
1750  // new buckets container, or to the first available spot in the new
1751  // bucket in the new buckets container.
1752  for (size_type old_bucket_slot = 0; old_bucket_slot < slot_per_bucket();
1753  ++old_bucket_slot) {
1754  if (!old_bucket.occupied(old_bucket_slot)) {
1755  continue;
1756  }
1757  const hash_value hv = hashed_key(old_bucket.key(old_bucket_slot));
1758  const size_type old_ihash = index_hash(old_hp, hv.hash);
1759  const size_type old_ahash = alt_index(old_hp, hv.partial, old_ihash);
1760  const size_type new_ihash = index_hash(new_hp, hv.hash);
1761  const size_type new_ahash = alt_index(new_hp, hv.partial, new_ihash);
1762  size_type dst_bucket_ind, dst_bucket_slot;
1763  if ((old_bucket_ind == old_ihash && new_ihash == new_bucket_ind) ||
1764  (old_bucket_ind == old_ahash && new_ahash == new_bucket_ind)) {
1765  // We're moving the key to the new bucket
1766  dst_bucket_ind = new_bucket_ind;
1767  dst_bucket_slot = new_bucket_slot++;
1768  } else {
1769  // We're moving the key to the old bucket
1770  assert((old_bucket_ind == old_ihash && new_ihash == old_ihash) ||
1771  (old_bucket_ind == old_ahash && new_ahash == old_ahash));
1772  dst_bucket_ind = old_bucket_ind;
1773  dst_bucket_slot = old_bucket_slot;
1774  }
1775  new_buckets.setKV(dst_bucket_ind, dst_bucket_slot++,
1776  old_bucket.partial(old_bucket_slot),
1777  old_bucket.movable_key(old_bucket_slot),
1778  std::move(old_bucket.mapped(old_bucket_slot)));
1779  }
1780  }

1806  // When we expand the contanier, we may need to expand the locks array, if
1807  // the current locks array is smaller than the maximum size and also smaller
1808  // than the number of buckets in the upcoming buckets container. In this
1809  // case, we grow the locks array to the smaller of the maximum lock array
1810  // size and the bucket count. This is done by allocating an entirely new lock
1811  // container, taking all the locks, copying over the counters, and then
1812  // finally adding it to the end of `all_locks_`, thereby designating it the
1813  // "current" locks container. It is the responsibility of the caller to
1814  // unlock all locks taken, including the new locks, whenever it is done with
1815  // them, so that old threads can resume and potentially re-start.
1816  void maybe_resize_locks(size_type new_bucket_count) {
1817  locks_t &current_locks = get_current_locks();
1818  if (!(current_locks.size() < kMaxNumLocks &&
1819  current_locks.size() < new_bucket_count)) {
1820  return;
1821  }
1822 
1823  locks_t new_locks(std::min(size_type(kMaxNumLocks), new_bucket_count),
1824  spinlock(), get_allocator());
1825  assert(new_locks.size() > current_locks.size());
1826  std::copy(current_locks.begin(), current_locks.end(), new_locks.begin());
1827  for (spinlock &lock : new_locks) {
1828  lock.lock();
1829  }
1830  all_locks_.emplace_back(std::move(new_locks));
1831  }

1833  // cuckoo_expand_simple will resize the table to at least the given
1834  // new_hashpower. When we're shrinking the table, if the current table
1835  // contains more elements than can be held by new_hashpower, the resulting
1836  // hashpower will be greater than `new_hp`. It needs to take all the bucket
1837  // locks, since no other operations can change the table during expansion.
1838  // Throws maximum_hashpower_exceeded if we're expanding beyond the
1839  // maximum hashpower, and we have an actual limit.
1840  template <typename TABLE_MODE, typename AUTO_RESIZE>
1841  cuckoo_status cuckoo_expand_simple(size_type new_hp) {
1842  auto all_locks_manager = lock_all(TABLE_MODE());
1843  const size_type hp = hashpower();
1844  cuckoo_status st = check_resize_validity<AUTO_RESIZE>(hp, new_hp);
1845  if (st != ok) {
1846  return st;
1847  }
1848 
1849  // Finish rehashing any data into buckets_.
1850  rehash_with_workers();
1851 
1852  // Creates a new hash table with hashpower new_hp and adds all the elements
1853  // from buckets_ and old_buckets_. Allow this map to spawn extra threads if
1854  // it needs to resize during the resize.
1855  cuckoohash_map new_map(hashsize(new_hp) * slot_per_bucket(),
1856  hash_function(), key_eq(), get_allocator());
1857  new_map.max_num_worker_threads(max_num_worker_threads());
1858 
1859  parallel_exec(
1860  0, hashsize(hp),
1861  [this, &new_map]
1862  (size_type i, size_type end, std::exception_ptr &eptr) {
1863  try {
1864  for (; i < end; ++i) {
1865  auto &bucket = buckets_[i];
1866  for (size_type j = 0; j < slot_per_bucket(); ++j) {
1867  if (bucket.occupied(j)) {
1868  new_map.insert(bucket.movable_key(j),
1869  std::move(bucket.mapped(j)));
1870  }
1871  }
1872  }
1873  } catch (...) {
1874  eptr = std::current_exception();
1875  }
1876  });
1877 
1878  // Finish rehashing any data in new_map.
1879  new_map.rehash_with_workers();
1880

1882  // have all the locks, so nobody else should be reading from the buckets
1883  // array. Then the old buckets will be deleted when new_map is deleted.
1884  maybe_resize_locks(new_map.bucket_count());
1885  buckets_.swap(new_map.buckets_);
1886 
1887  return ok;
1888  }

1909  start += work_per_thread;

1912  for (std::thread &t : threads) {
1913  t.join();
1914  }
